\begin{frame}{Bin Packing Problem}
\begin{columns}[T]
\begin{column}{0.52\textwidth}
\textbf{Given:} items with sizes $s_i$, bins with capacity $C$

\begin{frame}{Compact Formulation}
\textbf{Variables:} $x_{ib} \in \{0,1\}$ (item $i$ in bin $b$),
$y_b \in \{0,1\}$ (bin $b$ used)

\begin{align*}
  \min \quad & \sum_{b} y_b \\
  \text{s.t.} \quad
    & \sum_{b} x_{ib} = 1 && \forall\, i \quad \text{(assign each item)} \\
    & \sum_{i} s_i\, x_{ib} \le C\, y_b && \forall\, b \quad \text{(capacity)} \\
    & x_{ib}, y_b \in \{0,1\}
\end{align*}

% ---------------------------------------------------------
% Frame 6: Integer Master Problem
% ---------------------------------------------------------
\begin{frame}{Integer Master Problem}
For all feasible patterns $\mathcal{P}$, let $a_i^p = 1$ if item $i$ is in pattern $p$:

\begin{align*}
  \min \quad & \sum_{p \in \mathcal{P}} z_p \\
  \text{s.t.} \quad
    & \sum_{p \in \mathcal{P}} a_i^p\, z_p = 1 && \forall\, i \in \mathcal{I} \quad (\pi_i) \\
    & z_p \in \{0,1\} && \forall\, p \in \mathcal{P}
\end{align*}

\begin{itemize}
  \item Exponentially many patterns $\Rightarrow$ cannot enumerate all of $\mathcal{P}$
  \item \textbf{Stronger LP relaxation} than compact formulation
  \item Solution: \textbf{Branch-and-Price!}
\end{itemize}
\end{frame}

% Frame: LP Duality & Reduced Costs
% ---------------------------------------------------------
\begin{frame}{LP Duality \& Reduced Costs}
Consider a general LP and its dual:

\begin{columns}[T]
\begin{column}{0.44\textwidth}
\textbf{Primal}
\vspace{-0.2cm}
\begin{align*}
  \min \quad & c^\top x \\
  \text{s.t.} \quad & Ax = b \\
  & x \ge 0
\end{align*}
\end{column}
\begin{column}{0.44\textwidth}
\textbf{Dual}
\vspace{-0.2cm}
\begin{align*}
  \max \quad & b^\top \pi \\
  \text{s.t.} \quad & A^\top \pi \le c
\end{align*}
\end{column}
\end{columns}

\medskip
The \textbf{reduced cost} of variable $j$ measures its ``attractiveness'':
\[
  \bar{c}_j \;=\; c_j \;-\; \pi^\top a_j
\]

\begin{itemize}
  \item $\bar{c}_j < 0$: adding $x_j$ to the basis can \textbf{improve} the objective
  \item $\bar{c}_j \ge 0 \;\;\forall j$: current solution is \textbf{LP optimal}
        (complementary slackness)
\end{itemize}

\medskip
\textbf{Simplex method:} at each iteration, pivot on a variable with $\bar{c}_j < 0$.

When none exists $\Rightarrow$ optimality certificate.
\end{frame}

% ---------------------------------------------------------
% Frame: Why Column Generation Works
% ---------------------------------------------------------
\begin{frame}{Why Column Generation Works}
\textbf{Key insight:} to certify LP optimality, we only need to verify
$\bar{c}_j \ge 0$ for \emph{all} variables --- but we don't need them all in the LP!

\medskip
\begin{center}
\begin{tikzpicture}[
  box/.style={draw, rounded corners, fill=blue!10, minimum width=3.2cm,
              minimum height=0.8cm, font=\small\bfseries, align=center},
  arr/.style={-{Stealth}, thick},
  note/.style={font=\scriptsize, text=black!70, align=center}
]
  \node[box] (solve) at (0,0) {Solve RMP\\{\scriptsize(small LP)}};
  \node[box] (price) at (6,0) {Pricing oracle\\{\scriptsize$\min_j \;\bar{c}_j$}};

  \draw[arr, transform canvas={yshift=5mm}]
    (solve) -- node[above, font=\scriptsize] {dual values $\pi$} (price);
  \draw[arr, transform canvas={yshift=-5mm}]
    (price) -- node[below, font=\scriptsize] {column with $\bar{c}_j < 0$ (or ``none'')} (solve);

  \node[note] at (0, -1.6) {Only keeps a \emph{subset}\\of all variables};
  \node[note] at (6, -1.6) {Searches \emph{implicitly} over\\all (exponentially many) variables};
\end{tikzpicture}
\end{center}

\medskip
\begin{itemize}
  \item If the pricing oracle returns $\min_j \bar{c}_j \ge 0$:
        the RMP solution is optimal for the \textbf{full} LP
  \item If $\min_j \bar{c}_j < 0$: add the column and re-solve
  \item We never need to enumerate all columns --- just a \textbf{smart search}
\end{itemize}

\smallskip
\textbf{Dantzig--Wolfe:} exploit problem structure so that the pricing
oracle is a \emph{tractable} optimization problem (e.g., knapsack, shortest path).
\end{frame}

% Frame 9: RMP and Initial Columns

\begin{frame}{Restricted Master Problem}
The RMP uses only a \textbf{subset} $\mathcal{P}' \subseteq \mathcal{P}$ of patterns:

\begin{align*}
  \min \quad & \sum_{p \in \mathcal{P}'} z_p \\
  \text{s.t.} \quad
    & \sum_{p \in \mathcal{P}'} a_i^p\, z_p = 1 && \forall\, i \in \mathcal{I} \quad (\pi_i) \\
    & 0 \le z_p \le 1 && \forall\, p \in \mathcal{P}'
\end{align*}

\medskip
\textbf{Initial columns:} We need the RMP to be feasible from the start.

Simplest option: one item per pattern (each item alone in a bin).

% Frame 11: Pricing
% ---------------------------------------------------------
\begin{frame}{Pricing Problem}
Reduced cost of a new pattern $a$:\quad
$\bar{c} = \underbrace{1}_{\text{obj.\ coeff.}} - \underbrace{\sum_{i} a_i \pi_i}_{\text{col.\ coeffs.\ $\times$ duals}}$

\smallskip
Minimize reduced cost subject to bin capacity:
\vspace{-0.3cm}
\begin{align*}
  \min \quad & 1 - \sum_{i} a_i\, \pi_i \\
  \text{s.t.} \quad
    & \sum_{i} s_i\, a_i \le C, \quad a_i \in \{0,1\}
\end{align*}

\vspace{-0.3cm}
\pause
$= 1 - \max \sum_i a_i \pi_i \;\;\Rightarrow\;\;$ \textbf{0--1 Knapsack!}

\smallskip
When optimal reduced cost $\ge 0$: no improving pattern exists $\Rightarrow$ LP solved.
\end{frame}

% Frame 16: Why Standard Branching Fails
% ---------------------------------------------------------
\begin{frame}{Why Standard Branching Fails}
Standard variable branching on $z_p$:

\medskip
\begin{columns}[T]
\begin{column}{0.48\textwidth}
\begin{block}{$z_p = 1$}
Force \emph{one specific} packing out of exponentially many.

$\Rightarrow$ \textbf{Very restrictive}
\end{block}
\end{column}
\begin{column}{0.48\textwidth}
\begin{block}{$z_p = 0$}
Forbid \emph{one specific} packing out of exponentially many.

$\Rightarrow$ \textbf{Barely restrictive}
\end{block}
\end{column}
\end{columns}

% Frame 17: Ryan-Foster Branching
% ---------------------------------------------------------
\begin{frame}{Ryan-Foster Branching}
Branch on \textbf{pairs of items} $(i, j)$ instead of individual variables:

\smallskip
\begin{columns}[T]
\begin{column}{0.48\textwidth}
\begin{block}{Together}
Items $i$ and $j$ must be in the \textbf{same} bin.
\end{block}
\end{column}
\begin{column}{0.48\textwidth}
\begin{block}{Apart}
Items $i$ and $j$ must be in \textbf{different} bins.
\end{block}
\end{column}
\end{columns}

\smallskip
\begin{columns}[c]
\begin{column}{0.50\textwidth}
\textbf{How to find a pair:}
\begin{itemize}\setlength{\itemsep}{2pt}
  \item Compute implicit pair variable:\\[2pt]
        $x_{ij} = \sum_{p:\, i \in p,\, j \in p} z_p$
  \item Choose pair where $0 < x_{ij} < 1$ (fractional)
\end{itemize}

\smallskip
$\Rightarrow$ Much more \textbf{balanced} search tree
\end{column}
\begin{column}{0.47\textwidth}
\centering
\includegraphics[width=\linewidth]{media/ryan_foster.png}
\end{column}
\end{columns}
\end{frame}

  \item Scales to 100+ items
\end{itemize}
\end{column}

\begin{frame}{Branch-Price-and-Cut}
Strengthen branch-and-price by adding \textbf{cutting planes}.

\medskip
\textbf{Subset-row inequalities} for set partitioning:

For any triple of items $S = \{i, j, k\}$, consider all packings covering
$\ge 2$ items in $S$. If their LP values sum $> 1$:
\[
  \sum_{p:\, |p \cap S| \ge 2} z_p \le 1
\]

\medskip
\textbf{Separator plugin} in SCIP:
\begin{itemize}
  \item \texttt{sepaexeclp}: search for violated inequalities at each LP solution
  \item Return \texttt{SEPARATED} if cut added, \texttt{DIDNOTFIND} otherwise
\end{itemize}
\end{frame}
