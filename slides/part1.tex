model.setParam("limits/time", 60)       # time limit (seconds)

\end{lstlisting}

\medskip
\textbf{Emphasis settings} adjust groups of parameters at once:
\begin{lstlisting}
model.setEmphasis("OPTIMALITY")  # focus on proving optimality
model.setEmphasis("FEASIBILITY") # focus on finding solutions
\end{lstlisting}

\medskip
\textbf{Reading models from file:}
\begin{lstlisting}

model.readProblem("instance.mps")  # or .lp

\begin{frame}{Transportation Problem}
\begin{columns}[T]
\begin{column}{0.45\textwidth}
\begin{center}
\begin{tikzpicture}[
  sup/.style={circle,draw,fill=blue!15,minimum size=0.7cm,
              font=\scriptsize\bfseries},
  cus/.style={circle,draw,fill=red!15,minimum size=0.7cm,
              font=\scriptsize\bfseries},
  every edge/.style={draw,-{Stealth},thin},
]
  \node[sup] (s1) at (0, 1.5) {$S_1$};
  \node[sup] (s2) at (0, 0)   {$S_2$};
  \node[cus] (c1) at (3, 2.2) {$C_1$};
  \node[cus] (c2) at (3, 0.8) {$C_2$};
  \node[cus] (c3) at (3,-0.6) {$C_3$};

  \foreach \s in {s1,s2}
    \foreach \c in {c1,c2,c3}
      \draw[->] (\s) -- (\c);
\end{tikzpicture}
\end{center}
\end{column}
\begin{column}{0.55\textwidth}
\textbf{Sets:} suppliers $I$, customers $J$

\textbf{Parameters:} supply $s_i$, demand $d_j$, cost $c_{ij}$

\textbf{Variables:} $x_{ij} \ge 0$ (flow from $i$ to $j$)

\begin{align*}
  \min \quad & \sum_{i \in I}\sum_{j \in J} c_{ij}\, x_{ij} \\
  \text{s.t.} \quad
    & \sum_{j \in J} x_{ij} \le s_i  & \forall\, i \in I \\
    & \sum_{i \in I} x_{ij} \ge d_j  & \forall\, j \in J \\
    & x_{ij} \ge 0
\end{align*}
\end{column}
\end{columns}
\end{frame}

# Data as dictionaries
supply   = {"S1": 100, "S2": 150}
demand   = {"C1": 80, "C2": 90, "C3": 70}
cost     = {("S1","C1"): 4, ("S1","C2"): 6, ("S1","C3"): 9,
            ("S2","C1"): 5, ("S2","C2"): 3, ("S2","C3"): 7}

# Variables via nested comprehension
x = {(i,j): model.addVar(name=f"x_{i}_{j}", vtype="C")
     for i in supply for j in demand}

# Constraints via loops
for i in supply:
    model.addCons(
        quicksum(x[i,j] for j in demand) <= supply[i])
for j in demand:
    model.addCons(
        quicksum(x[i,j] for i in supply) >= demand[j])
\end{lstlisting}
\end{frame}

\begin{column}{0.57\textwidth}
\textbf{Classic 0--1 Knapsack:} select items to maximize value
without exceeding capacity.

\smallskip
{\small\textbf{Applications:} budgeting, cargo loading, resource allocation.}

\smallskip
\textbf{Variables:} $x_j \in \{0,1\}$ (take item $j$?)

\begin{align*}
  \max \quad & \sum_{j=1}^{n} v_j \, x_j \\
  \text{s.t.} \quad
    & \sum_{j=1}^{n} w_j \, x_j \le W \\
    & x_j \in \{0,1\} & \forall\, j
\end{align*}

{\small where $v_j$ = value, $w_j$ = weight, $W$ = capacity.}

\end{column}
\end{columns}
\end{frame}

\end{tikzpicture}

\smallskip
{\scriptsize\color{red!70!black} Tour in red;
\color{gray!60} complete graph in gray}
\end{column}
\begin{column}{0.55\textwidth}
\textbf{Goal:} Shortest tour visiting every city exactly once.

\smallskip
\textbf{MTZ formulation:} position variables prevent subtours.

\begin{align*}
  \min \quad & \sum_{i \ne j} d_{ij}\, x_{ij} \\
  \text{s.t.} \quad
    & \textstyle\sum_{j \ne i} x_{ij} = 1
      & \forall\, i \\
    & \textstyle\sum_{i \ne j} x_{ij} = 1
      & \forall\, j \\
    & u_i - u_j + n\, x_{ij} \le n{-}1
      & \forall\, i,j \ne 0 \\
    & x_{ij} \in \{0,1\}
\end{align*}

\begin{itemize}\setlength\itemsep{0pt}
  \item $O(n^2)$ variables and constraints. Easy to implement.
  \item Weak LP relaxation. Part 2 shows a stronger approach.
\end{itemize}
\end{column}
\end{columns}
\end{frame}

\begin{column}{0.60\textwidth}
\textbf{Given:} undirected graph $G=(V,E)$, set of colors $K=\{1,\dots,k\}$.

\textbf{Goal:} minimize the number of colors used such that
no two adjacent vertices share a color.

\smallskip
{\small\textbf{Variables:} $x_{v,c} \in \{0,1\}$ (vertex $v$ gets color $c$?), \;
$w_c \in \{0,1\}$ (color $c$ used?)}

\begin{align*}
  \min \quad & \sum_{c \in K} w_c \\
  \text{s.t.} \quad
    & \sum_{c \in K} x_{v,c} = 1
      & \forall\, v \in V \\
    & x_{u,c} + x_{v,c} \le w_c
      & \forall\, (u,v) \in E,\; c \in K \\
    & x_{v,c},\, w_c \in \{0,1\}
\end{align*}

{\small
\textbf{Symmetry:} permuting colors $\Rightarrow$ equivalent solution.\\
\textbf{Break:} $w_1 \ge w_2 \ge \cdots \ge w_k$.}
\end{column}

\begin{frame}{Nonlinear Blending (Pooling Problem)}
\begin{columns}[T]
\begin{column}{0.40\textwidth}
\begin{center}
\begin{tikzpicture}[scale=0.9,
  src/.style={circle,draw,fill=blue!15,minimum size=0.7cm,
              font=\scriptsize\bfseries},
  pool/.style={rectangle,draw,rounded corners=4pt,fill=yellow!20,
               minimum width=1.2cm,minimum height=0.7cm,
               font=\scriptsize\bfseries},
  prod/.style={circle,draw,fill=red!15,minimum size=0.7cm,
               font=\scriptsize\bfseries},
  arr/.style={-{Stealth},thin},
]
  % Sources
  \node[src]  (s1) at (0, 2.2) {$A$};
  \node[src]  (s2) at (0, 0.4) {$B$};

  % Pool
  \node[pool] (pl) at (2.2, 1.3) {Pool};

  % Products
  \node[prod] (p1) at (4.4, 2.2) {$X$};
  \node[prod] (p2) at (4.4, 0.4) {$Y$};

  % Source -> Pool
  \draw[arr] (s1) -- node[above,font=\tiny]{$x_A$} (pl);
  \draw[arr] (s2) -- node[below,font=\tiny]{$x_B$} (pl);

  % Pool -> Products
  \draw[arr] (pl) -- node[above,font=\tiny]{$y_X$} (p1);
  \draw[arr] (pl) -- node[below,font=\tiny]{$y_Y$} (p2);

  % Direct bypass arcs (wider bend to clear the pool)
  \draw[arr,densely dashed,gray] (s1) to[bend left=30]
    node[above,font=\tiny,pos=0.5]{$z_{AX}$} (p1);
  \draw[arr,densely dashed,gray] (s2) to[bend right=30]
    node[below,font=\tiny,pos=0.5]{$z_{BY}$} (p2);

  % Legend
  \node[font=\tiny,gray] at (2.2,-0.9) {dashed = direct bypass};
\end{tikzpicture}
\end{center}
\end{column}
\begin{column}{0.60\textwidth}
Blend raw materials through a mixing pool to make products with
quality specifications.

\smallskip
{\small\textbf{Parameters:} cost $c_s$, quality $q_s$ per source;
revenue $r_p$, max quality $\bar{q}_p$ per product}

{\small\textbf{Variables:} $x_s$ (source to pool), $y_p$ (pool to product),
$z_{sp}$ (direct bypass), $\lambda$ (pool quality)}

\begin{align*}
  \max \quad & \textstyle\sum_p r_p\,d_p - \sum_s c_s\,(\,x_s + \sum_p z_{sp})\\[3pt]
  \text{s.t.} \quad
    & \textstyle\sum_s x_s = \sum_p y_p
      & \text{(balance)} \\
    & \lambda \textstyle\sum_s x_s = \sum_s q_s\, x_s
      & \text{(pool qual.)} \\
    & \lambda\, y_p + \textstyle\sum_s q_s\, z_{sp} \le \bar{q}_p\, d_p
      & \forall\, p
\end{align*}
where $d_p = y_p + \sum_s z_{sp}$.

\smallskip
{\small\textbf{Note:} The terms $\lambda\, x_s$ and $\lambda\, y_p$ are
\textbf{bilinear}. This is a nonconvex NLP that SCIP can solve globally.}
\end{column}
\end{columns}
\end{frame}

\textbf{Given:} generators with on/off decisions, min/max output, and costs.

\textbf{Key constraint:} if on ($y_i{=}1$), output satisfies

    & y_i = 1 \;\Rightarrow\; P_i^{\min} \le p_i \le P_i^{\max}
\end{align*}

\begin{frame}[fragile]{Indicator Constraints}
Big-M constraints model logical implications using a large constant $M$.
They work, but choosing $M$ and the resulting LP relaxation can be tricky.

\medskip
\textbf{Indicator constraints} link a binary variable directly to a
linear constraint, no $M$ needed:

\medskip
\begin{center}
$y = 1 \;\Rightarrow\; \text{constraint holds}$
\end{center}

\medskip
\begin{columns}[T]
\begin{column}{0.48\textwidth}
\textbf{Big-M:}
\begin{align*}
  p_i &\le P_i^{\max} \cdot y_i \\
  p_i &\ge P_i^{\min} \cdot y_i
\end{align*}
\end{column}
\begin{column}{0.48\textwidth}
\textbf{Indicator:}
\begin{lstlisting}
addConsIndicator(p[i] >= P_min[i], y[i])
\end{lstlisting}
\end{column}
\end{columns}

\medskip
\begin{itemize}
  \item SCIP handles the logic internally (branching, propagation).
  \item Often tighter LP relaxation and faster solving.
\end{itemize}
\end{frame}
