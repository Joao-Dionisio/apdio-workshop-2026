\section{Out-of-the-box Solver}

\begin{frame}{SCIP as a Black-Box Solver}
  \begin{itemize}
  \item SCIP is a solver for finding globally optimal solutions of
    \begin{itemize}
    \item mixed-integer (nonlinear) programs
    \item satisfiability problems
    \item constraint integer programs
    \end{itemize}
  \item supports exact arithmetic for mixed-integer linear programs
  \end{itemize}
\end{frame}

\begin{frame}{Interfaces}
  SCIP has interfaces to many popular programming languages
  \begin{itemize}
  \item C/C++
  \item Python
  \item Julia
  \item Rust
  \item Matlab
  \item Java
  \end{itemize}
\end{frame}

\begin{frame}[fragile]{SCIP is Easy to Use}
  \begin{itemize}
  \item executable can read many popular file formats (mps, lp, etc.)
  \item modeling problems via SCIP's interfaces is very easy
  \end{itemize}
  \begin{lstlisting}
from pyscipopt import Model

model = Model()

x = model.addVar("x", vtype="C", lb=0)
y = model.addVar("y", vtype="I", lb=0)

model.addCons(x + 2*y <= 10)
model.setObjective(x + y, "maximize")

model.optimize()
  \end{lstlisting}
\end{frame}

\section{Branch-and-Price}

\begin{frame}{Branch-and-Price}
  required functionality
  \begin{itemize}
  \item column generation
  \item embedding of column generation in branch-and-bound
  \item custom branching rules
  \end{itemize}

  \vfill

  SCIP naturally supports these features by
  \begin{itemize}
  \item plugin for generating new columns (\plugin{pricer})
  \item automatic handling of branch-and-bound tree including warm start and heuristics
  \item plugin for custom branching rules (\plugin{branching rule})
  \end{itemize}

  \vfill
\end{frame}

\section{Separation}

\begin{frame}{Cutting Planes}
  required functionality
  \begin{itemize}
  \item access to solution
  \item creating and adding custom cutting planes
  \end{itemize}

  \vfill

  SCIP naturally supports these features by
  \begin{itemize}
  \item plugin for custom cutting planes (\plugin{separator})
  \item automatic embedding in branch-and-bound, including cut storage and selection

\section{Research}

\begin{frame}{Two Sides of Research}
  \textbf{Research on Modeling}
  \begin{itemize}
  \item how to formulate a problem as a mathematical program
  \item comparing formulations: strength, size, solvability
  \item developing valid inequalities and reformulation techniques
  \end{itemize}

  \vfill

  \textbf{Research on Optimization}
  \begin{itemize}
  \item how to solve mathematical programs more efficiently
  \item new branching, cutting, presolving, or decomposition strategies
  \item algorithm design, complexity, and computational experiments
  \end{itemize}
\end{frame}

\begin{frame}{SCIP Enables Research on Optimization}
  \begin{itemize}
  \item plugin architecture allows implementing and testing new ideas within a mature solver
  \item full access to solving process internals (bounds, LP solutions, tree structure)
  \item benchmark new methods against state-of-the-art techniques
  \item open-source: inspect, modify, and extend any component
  \end{itemize}
\end{frame}
